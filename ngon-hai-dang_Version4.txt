import * as THREE from 'three';
import { PointerLockControls } from 'three/addons/controls/PointerLockControls.js';

/* ============================================================
   NGỌN HẢI ĐĂNG — combined: top-door + improved sky & ocean
   ============================================================ */

const EYE_HEIGHT = 1.68;
const WALK_SPEED = 3.4;
const RUN_SPEED = 6.2;

const TOWER_R = 4.5;         // outer tower radius
const FLOOR_R = 4.15;        // floor disc radius
const STAIR_R = 3.15;        // radius of the spiral staircase ring
const GROUND_Y = 0;
const F1_Y = 5.3, F2_Y = 10.7, F3_Y = 16.1, TOP_Y = 21.5;
const TOTAL_TURNS = 3.1;
const TOTAL_ANGLE = TOTAL_TURNS * Math.PI * 2;
const NOTCH_HALF = 0.34; // radians

function angleAtHeight(y){ return (y / TOP_Y) * TOTAL_ANGLE; }

/* ---------- renderer / scene / camera ---------- */
const canvasHolder = document.body;
const renderer = new THREE.WebGLRenderer({ antialias: true, powerPreference: 'high-performance' });
renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
renderer.setSize(window.innerWidth, window.innerHeight);
// improved color management & lighting
renderer.outputColorSpace = THREE.SRGBColorSpace;
renderer.toneMapping = THREE.ACESFilmicToneMapping;
renderer.toneMappingExposure = 1.05;
renderer.physicallyCorrectLights = true;
renderer.shadowMap.enabled = true;
renderer.shadowMap.type = THREE.PCFSoftShadowMap;
canvasHolder.appendChild(renderer.domElement);

const scene = new THREE.Scene();
const FOG_COLOR = new THREE.Color(0xcfe0e8);
scene.fog = new THREE.FogExp2(FOG_COLOR, 0.0048);

const camera = new THREE.PerspectiveCamera(72, window.innerWidth / window.innerHeight, 0.05, 3000);
camera.position.set(0, EYE_HEIGHT, 8.5);

/* ---------- lights ---------- */
const sunDir = new THREE.Vector3(0.45, 0.42, -0.55).normalize();
const hemi = new THREE.HemisphereLight(0xfbe8c6, 0x142230, 0.9);
scene.add(hemi);
const sun = new THREE.DirectionalLight(0xffd9a8, 1.7);
sun.position.copy(sunDir).multiplyScalar(200);
sun.castShadow = true;
sun.shadow.mapSize.set(2048, 2048);
sun.shadow.camera.near = 80;
sun.shadow.camera.far = 320;
sun.shadow.camera.left = -26;
sun.shadow.camera.right = 26;
sun.shadow.camera.top = 26;
sun.shadow.camera.bottom = -26;
sun.shadow.bias = -0.0015;
sun.shadow.camera.updateProjectionMatrix();
scene.add(sun);
scene.add(sun.target);
const fillLight = new THREE.DirectionalLight(0x6fa8c9, 0.35);
fillLight.position.set(-100, 80, 120);
scene.add(fillLight);
// soft light standing in for moonlight at night
const moonLight = new THREE.DirectionalLight(0x8fb6e0, 0);
moonLight.position.set(-45, 60, 55);
scene.add(moonLight);

/* ============================================================
   DAY / NIGHT CYCLE STATE
   ============================================================ */
const DAY_LENGTH = 600; // seconds for one full cycle
let dayPhase = 0.24;    // start just after sunrise (buổi sáng)
const AZ_X = 0.634, AZ_Z = -0.775; // fixed sun azimuth (from original lighting direction)
const SKY_DAY_TOP = new THREE.Color(0x2b5a8a);
const SKY_DAY_HORIZON = new THREE.Color(0xf3c98d);
const SKY_DAY_HORIZONLOW = new THREE.Color(0xffe3b0);
const SKY_NIGHT_TOP = new THREE.Color(0x040814);
const SKY_NIGHT_HORIZON = new THREE.Color(0x0e1c34);
const SKY_NIGHT_HORIZONLOW = new THREE.Color(0x16283f);
const DUSK_TINT = new THREE.Color(0xff7a4d);
const FOG_DAY = new THREE.Color(0xcfe0e8);
const FOG_NIGHT = new THREE.Color(0x0a1220);
const OCEAN_DAY_DEEP = new THREE.Color(0x08344a), OCEAN_NIGHT_DEEP = new THREE.Color(0x020a12);
const OCEAN_DAY_SHALLOW = new THREE.Color(0x2f92a3), OCEAN_NIGHT_SHALLOW = new THREE.Color(0x0a2530);
const OCEAN_DAY_FOAM = new THREE.Color(0xdcefe9), OCEAN_NIGHT_FOAM = new THREE.Color(0x7d97a3);
const _tmpCol = new THREE.Color();

/* ============================================================
   IMPROVED SKY DOME (gradient + atmospheric tint + dynamic clouds)
   ============================================================ */
const skyGeo = new THREE.SphereGeometry(1500, 36, 24);
const skyMat = new THREE.ShaderMaterial({
  side: THREE.BackSide,
  fog: false,
  uniforms: {
    uSunDir: { value: sunDir },
    uTop: { value: new THREE.Color(0x2b5a8a) },
    uHorizon: { value: new THREE.Color(0xf3c98d) },
    uHorizonLow: { value: new THREE.Color(0xffe3b0) },
    uTime: { value: 0.0 },
    uCloudScale: { value: 0.7 },
    uCloudSpeed: { value: 0.03 },
    uCloudContrast: { value: 1.1 },
    uSunIntensity: { value: 1.0 }
  },
  vertexShader: `
    varying vec3 vWorldDir;
    varying vec3 vPosition;
    void main(){
      vWorldDir = normalize(position);
      vPosition = position;
      gl_Position = projectionMatrix * modelViewMatrix * vec4(position,1.0);
    }
  `,
  fragmentShader: `
    varying vec3 vWorldDir;
    varying vec3 vPosition;
    uniform vec3 uSunDir;
    uniform vec3 uTop, uHorizon, uHorizonLow;
    uniform float uTime, uCloudScale, uCloudSpeed, uCloudContrast, uSunIntensity;

    // --- hash / noise / fbm (2D) ---
    float hash(vec2 p){ p = 50.0 * fract(p * 0.3183099 + vec2(0.71,0.113)); return fract(p.x * p.y * 95.4307); }
    float noise(vec2 p){
      vec2 i = floor(p);
      vec2 f = fract(p);
      float a = hash(i);
      float b = hash(i + vec2(1.0, 0.0));
      float c = hash(i + vec2(0.0, 1.0));
      float d = hash(i + vec2(1.0, 1.0));
      vec2 u = f*f*(3.0-2.0*f);
      return mix(a,b,u.x) + (c-a)*u.y*(1.0-u.x) + (d-b)*u.x*u.y;
    }
    float fbm(vec2 p){
      float v=0.0, a=0.5;
      for(int i=0;i<5;i++){
        v += a*noise(p);
        p = p*2.0 + vec2(1.7, -0.6);
        a *= 0.5;
      }
      return v;
    }

    void main(){
      // base gradient by height (vWorldDir.y)
      float h = clamp(vWorldDir.y, -1.0, 1.0);
      vec3 base = mix(uHorizonLow, uHorizon, smoothstep(-0.02, 0.18, h));
      base = mix(base, uTop, smoothstep(0.1, 0.85, h));

      // sun disk contribution (soft)
      float sunAmt = max(dot(vWorldDir, normalize(uSunDir)), 0.0);
      vec3 sunCol = vec3(1.0,0.85,0.6) * pow(sunAmt, 420.0) * 2.2 * uSunIntensity;
      base += sunCol;

      // subtle Rayleigh-ish bluish tint (soft)
      float ray = pow(max(0.0, vWorldDir.y*0.5 + 0.5), 1.6);
      base = mix(base, base * vec3(0.95, 1.02, 1.12), 0.18 * ray);

      // clouds (fbm in spherical coords)
      vec2 cloudUV = vec2(atan(vWorldDir.z, vWorldDir.x), asin(clamp(vWorldDir.y,-1.0,1.0)));
      cloudUV *= vec2(0.5, 1.0); // scale
      cloudUV *= (1.0/uCloudScale);
      cloudUV += vec2(uTime * uCloudSpeed, uTime * uCloudSpeed * 0.3);
      float clouds = fbm(cloudUV * 2.5);
      clouds = smoothstep(0.45, 0.78, clouds * uCloudContrast);

      // soften clouds near sun (sun kills clouds a bit)
      clouds *= 1.0 - pow(sunAmt, 2.2) * 0.95;

      // cloud color: brighter near sun / whiter, darker in shadow
      vec3 cloudColor = mix(vec3(0.8,0.82,0.86), vec3(1.0,0.98,0.92), clamp(sunAmt*1.6,0.0,1.0));
      base = mix(base, cloudColor, clouds * 0.75);

      // horizon glow (warm)
      float horizonGlow = exp(-pow((1.0 - h) * 4.0, 1.6));
      base += vec3(1.0,0.6,0.35) * horizonGlow * 0.12;

      gl_FragColor = vec4(base, 1.0);
    }
  `
});
scene.add(new THREE.Mesh(skyGeo, skyMat));

/* ============================================================

const SKY_DIST = 900;

function makeMoonTexture(){
  const c = document.createElement('canvas'); c.width = c.height = 256;
  const ctx = c.getContext('2d');
  const g = ctx.createRadialGradient(96, 90, 4, 128, 128, 128);
  g.addColorStop(0, '#fbf6e8');
  g.addColorStop(0.55, '#e4dcc4');
  g.addColorStop(1, '#b9b096');
  ctx.fillStyle = g; ctx.beginPath(); ctx.arc(128,128,124,0,Math.PI*2); ctx.fill();
  ctx.fillStyle = 'rgba(150,142,120,0.35)';
  const craters = [[80,70,16],[150,60,10],[170,140,22],[95,160,13],[60,120,8],[190,190,15]];
  craters.forEach(([x,y,r])=>{ ctx.beginPath(); ctx.arc(x,y,r,0,Math.PI*2); ctx.fill(); });
  return new THREE.CanvasTexture(c);
}
const moonGroup = new THREE.Group();
const moonMat = new THREE.MeshBasicMaterial({ map: makeMoonTexture(), fog: false, transparent: true, opacity: 1 });
const moon = new THREE.Mesh(new THREE.SphereGeometry(16, 24, 24), moonMat);
moonGroup.add(moon);
scene.add(moonGroup);

const sunSpriteMat = new THREE.SpriteMaterial({ color: 0xfff3d0, transparent: true, depthWrite: false, fog: false, opacity: 0.95 });
const sunSprite = new THREE.Sprite(sunSpriteMat);
sunSprite.scale.set(46, 46, 1);
scene.add(sunSprite);

/* starfield — fades in at night */
const STAR_COUNT = 1200;
const starPos = new Float32Array(STAR_COUNT * 3);
for (let i=0;i<STAR_COUNT;i++){
  const r = 1400;
  const theta = Math.random()*Math.PI*2;
  const phi = Math.acos(Math.random()*0.92); // keep mostly upper hemisphere
  starPos[i*3]   = r*Math.sin(phi)*Math.cos(theta);
  starPos[i*3+1] = r*Math.cos(phi) + 40;
  starPos[i*3+2] = r*Math.sin(phi)*Math.sin(theta);
}
const starGeo = new THREE.BufferGeometry();
starGeo.setAttribute('position', new THREE.BufferAttribute(starPos, 3));
const starMat = new THREE.PointsMaterial({ color: 0xf5f6ff, size: 1.8, sizeAttenuation: false, transparent: true, opacity: 0, fog: false, depthWrite: false });
const stars = new THREE.Points(starGeo, starMat);
scene.add(stars);

/* ============================================================
   IMPROVED OCEAN SHADER (multiple wave layers, normals from derivatives, foam)
   ============================================================ */
const oceanGeo = new THREE.PlaneGeometry(3000, 3000, 220, 220);
oceanGeo.rotateX(-Math.PI/2);

const oceanMat = new THREE.ShaderMaterial({
  transparent: true,
  fog: false,
  uniforms: {
    uTime: { value: 0 },
    uSunDir: { value: sunDir },
    uDeep: { value: new THREE.Color(0x063149) },
    uShallow: { value: new THREE.Color(0x1f8aa0) },
    uFoam: { value: new THREE.Color(0xdfeef0) },
    uCamPos: { value: camera.position },
    uFogColor: { value: FOG_COLOR },
    uWaveAmp: { value: 1.0 },
    uChoppiness: { value: 1.0 },
    uSpecularPower: { value: 90.0 },
    uTimeScale: { value: 1.0 }
  },
  vertexShader: `
    uniform float uTime, uWaveAmp, uChoppiness, uTimeScale;
    varying vec3 vWorldPos;
    varying vec3 vNormal;
    varying float vWave;
    // Gerstner-ish wave parameters
    const int WAVE_COUNT = 6;
    vec2 dirs[WAVE_COUNT];
    float amps[WAVE_COUNT];
    float freqs[WAVE_COUNT];
    float speeds[WAVE_COUNT];
    float phases[WAVE_COUNT];
    void initWaves(){
      dirs[0] = vec2(1.0, 0.2); dirs[1] = vec2(0.6, 0.8); dirs[2] = vec2(-0.7, 0.4);
      dirs[3] = vec2(-0.2, -1.0); dirs[4] = vec2(0.9, -0.5); dirs[5] = vec2(-0.5, -0.7);
      amps[0]=0.36; amps[1]=0.18; amps[2]=0.14; amps[3]=0.08; amps[4]=0.06; amps[5]=0.04;
      freqs[0]=0.018; freqs[1]=0.033; freqs[2]=0.055; freqs[3]=0.072; freqs[4]=0.12; freqs[5]=0.18;
      speeds[0]=0.12; speeds[1]=0.18; speeds[2]=0.26; speeds[3]=0.32; speeds[4]=0.42; speeds[5]=0.6;
      phases[0]=0.0; phases[1]=1.2; phases[2]=0.8; phases[3]=2.0; phases[4]=1.4; phases[5]=2.7;
    }
    // compute height+normal via sum of Gerstner-like components
    void main(){
      initWaves();
      vec3 pos = position;
      float h = 0.0;
      vec3 n = vec3(0.0,1.0,0.0);
      for(int i=0;i<WAVE_COUNT;i++){
        vec2 d = normalize(dirs[i]);
        float freq = freqs[i];
        float amp = amps[i] * uWaveAmp;
        float phase = phases[i];
        float speed = speeds[i];
        float t = (uTime * uTimeScale) * speed;
        float w = dot(d, pos.xz) * freq + phase + t;
        float sinw = sin(w);
        float cosw = cos(w);
        // horizontal displacement for choppiness
        pos.x += d.x * (amp * uChoppiness * cosw);
        pos.z += d.y * (amp * uChoppiness * cosw);
        h += amp * sinw;
        // accumulate normal using approximate derivatives
        n.x += -d.x * amp * freq * cosw;
        n.y += 1.0;
        n.z += -d.y * amp * freq * cosw;
      }
      pos.y += h;
      n = normalize(n);
      vNormal = n;
      vWave = h;
      vec4 world = modelMatrix * vec4(pos,1.0);
      vWorldPos = world.xyz;
      gl_Position = projectionMatrix * viewMatrix * world;
    }
  `,
  fragmentShader: `
    uniform vec3 uSunDir, uDeep, uShallow, uFoam, uCamPos, uFogColor;
    uniform float uSpecularPower;
    varying vec3 vWorldPos;
    varying vec3 vNormal;
    varying float vWave;
    void main(){
      vec3 viewDir = normalize(uCamPos - vWorldPos);
      float fresnel = pow(1.0 - clamp(dot(normalize(vNormal), viewDir), 0.0, 1.0), 3.0);
      float depthFactor = clamp((vWorldPos.y + 1.2) / 2.8, 0.0, 1.0);
      vec3 base = mix(uDeep, uShallow, depthFactor);
      float slope = 1.0 - dot(normalize(vNormal), vec3(0.0,1.0,0.0));
      float foamAmt = smoothstep(0.04, 0.22, slope*2.0 + vWave*0.6);
      vec3 halfDir = normalize(normalize(uSunDir) + viewDir);
      float spec = pow(max(dot(normalize(vNormal), halfDir), 0.0), uSpecularPower);
      vec3 col = base;
      col += fresnel * vec3(0.45,0.6,0.68) * 0.6;
      col += spec * vec3(1.0,0.95,0.85) * 1.6;
      col = mix(col, uFoam, foamAmt);
      float d = length(uCamPos - vWorldPos);
      float fogF = 1.0 - exp(-0.0018 * d);
      col = mix(col, uFogColor, clamp(fogF, 0.0, 1.0));
      gl_FragColor = vec4(col, 0.96);
    }
  `
});

/* update door animations (lower + top) */
function updateDoor(delta){
  // lower door
  const target = doorOpen ? 1 : 0;
  doorAnim += (target - doorAnim) * Math.min(1, delta * 4.5);
  doorHingeL.rotation.y = doorAnim * DOOR_OPEN_ANGLE;
  doorHingeR.rotation.y = -doorAnim * DOOR_OPEN_ANGLE;

  // top door
  const targetTop = doorOpenTop ? 1 : 0;
  doorAnimTop += (targetTop - doorAnimTop) * Math.min(1, delta * 4.5);
  // open outward: positive rotation on left, negative on right (hinge local Y)
  doorHingeTopL.rotation.y = -a + doorAnimTop * DOOR_TOP_OPEN_ANGLE;
  doorHingeTopR.rotation.y = -a - doorAnimTop * DOOR_TOP_OPEN_ANGLE;
}

// blocks movement through the doorway while the doors are still mostly closed
function blockedByClosedDoor(x, z){
  // lower door check (original)
  if (doorAnim <= 0.4){
    if (Math.abs(x - DOOR_RADIUS) <= 0.3 && Math.abs(z) < DOOR_W/2 - 0.05) return true;
  }

  // top door check
  if (doorAnimTop <= 0.4){
    const px = x - center.x, pz = z - center.z;
    // project onto normalOut and perp
    const distAlongNormal = px * normalOut.x + pz * normalOut.z;
    if (Math.abs(distAlongNormal) <= 0.32){
      const lateral = px * perp.x + pz * perp.z;
      if (Math.abs(lateral) < openingWidth/2 - 0.05){
        const cy = center.y;
        if (Math.abs(camera.position.y - cy) < TOP_OPEN_H/2 + 0.6) return true;
      }
    }
  }

  return false;
}

function updateInteractables(t){
  const active = controls.isLocked || experienceActive;

  let nearestFood = null, nearestFoodDist = Infinity;
  foodCrates.forEach(crate=>{
    if (crate.userData.collected){
      if (t >= crate.userData.respawnAt){ crate.userData.collected = false; crate.visible = true; }
      return;
    }
    crate.userData.glow.intensity = 0.9 + Math.sin(t*2 + crate.userData.bobPhase)*0.3;
    if (!active) return;
    const dx = crate.position.x - camera.position.x;
    const dz = crate.position.z - camera.position.z;
    const dy = crate.position.y + 0.3 - camera.position.y;
    const dist = Math.sqrt(dx*dx+dy*dy+dz*dz);
    if (dist < PICKUP_RANGE && dist < nearestFoodDist){ nearestFood = crate; nearestFoodDist = dist; }
  });

  let doorDist = Infinity;
  let doorDistTop = Infinity;
  if (active){
    const ddx = DOOR_POS.x - camera.position.x;
    const ddy = DOOR_POS.y - camera.position.y;
    const ddz = DOOR_POS.z - camera.position.z;
    doorDist = Math.sqrt(ddx*ddx+ddy*ddy+ddz*ddz);

    const td = TOP_DOOR_POS.clone().sub(camera.position);
    doorDistTop = td.length();
  }
  const doorNear = doorDist < DOOR_INTERACT_RANGE;
  const doorNearTop = doorDistTop < DOOR_INTERACT_RANGE;
  const useFood = !!nearestFood && (!doorNear && !doorNearTop || nearestFoodDist <= Math.min(doorDist, doorDistTop));

  if (useFood){
    interactLabel.textContent = 'để lấy';
    interactBtnIcon.textContent = '🎒';
    interactBtnLabel.textContent = 'Lấy';
    interactPrompt.classList.add('show');
    if (interactRequested) collectFood(nearestFood);
  } else if (doorNearTop){
    interactLabel.textContent = doorOpenTop ? 'để đóng cửa' : 'để mở cửa';
    interactBtnIcon.textContent = '🚪';
    interactBtnLabel.textContent = doorOpenTop ? 'Đóng' : 'Mở';
    interactPrompt.classList.add('show');
    if (interactRequested) toggleDoor(true);
  } else if (doorNear){
    interactLabel.textContent = doorOpen ? 'để đóng cửa' : 'để mở cửa';
    interactBtnIcon.textContent = '🚪';
    interactBtnLabel.textContent = doorOpen ? 'Đóng' : 'Mở';
    interactPrompt.classList.add('show');
    if (interactRequested) toggleDoor(false);
  } else {
    interactPrompt.classList.remove('show');
  }
  interactRequested = false;
}

const JOY_RADIUS = 46;
let joystickTouchId = null;

function updateJoystick(touch){
  const rect = joystickBase.getBoundingClientRect();
  const cx = rect.left + rect.width/2, cy = rect.top + rect.height/2;
  let dx = touch.clientX - cx, dy = touch.clientY - cy;
  const dist = Math.hypot(dx, dy);
  if (dist > JOY_RADIUS){ dx = dx/dist*JOY_RADIUS; dy = dy/dist*JOY_RADIUS; }
  joystickKnob.style.transform = `translate(${dx}px, ${dy}px)`;
  touchMoveX = dx / JOY_RADIUS;
  touchMoveZ = -dy / JOY_RADIUS;
}

  touchMoveX = 0; touchMoveZ = 0;
  joystickKnob.style.transform = 'translate(0px, 0px)';
}

let feetY = 0, targetFeetY = 0;
let bobPhase = 0;
const raycaster = new THREE.Raycaster();
const DOWN = new THREE.Vector3(0,-1,0);

function sampleGround(x, z){
  raycaster.set(new THREE.Vector3(x, feetY + 14, z), DOWN);
  raycaster.far = 40;
  const hits = raycaster.intersectObjects(walkables, false);
  if (hits.length){
    const y = hits[0].point.y;
    if (Math.abs(y - feetY) < 2.4) return y;
  }
  return null;
}

let lastLabel = '';

function updateMovement(delta){
  if (!controls.isLocked && !experienceActive) return;
  const speed = (keys['ShiftLeft']||keys['ShiftRight']||touchRun) ? RUN_SPEED : WALK_SPEED;
  let vx = 0, vz = 0;
  if (keys['KeyW']||keys['ArrowUp']) vz += 1;
  if (keys['KeyS']||keys['ArrowDown']) vz -= 1;
  if (keys['KeyA']||keys['ArrowLeft']) vx -= 1;
  if (keys['KeyD']||keys['ArrowRight']) vx += 1;
  vx += touchMoveX;
  vz += touchMoveZ;
  const moving = (vx !== 0 || vz !== 0);
  if (moving){
    const len = Math.hypot(vx,vz);
    vx /= len; vz /= len;
  }

  const dist = speed * delta;

  let px = camera.position.x, pz = camera.position.z;
  controls.moveRight(vx * dist);
  const t1 = sampleGround(camera.position.x, camera.position.z);
  if (t1 === null || blockedByClosedDoor(camera.position.x, camera.position.z)) { camera.position.x = px; camera.position.z = pz; }

  px = camera.position.x; pz = camera.position.z;
  controls.moveForward(vz * dist);
  const t2 = sampleGround(camera.position.x, camera.position.z);
  if (t2 === null || blockedByClosedDoor(camera.position.x, camera.position.z)) { camera.position.x = px; camera.position.z = pz; }

  const finalY = sampleGround(camera.position.x, camera.position.z);
  if (finalY !== null) targetFeetY = finalY;
  feetY += (targetFeetY - feetY) * Math.min(1, delta * 9);

  bobPhase += delta * (moving ? (speed*2.1) : 4);
  const bob = moving ? Math.sin(bobPhase) * 0.045 : Math.sin(bobPhase)*0.006;

  camera.position.y = feetY + EYE_HEIGHT + bob;

  const label = currentLevelLabel();
  if (label !== lastLabel){ levelText.textContent = label; lastLabel = label; }
}

let audioCtx, noiseSrc, gainNode, muted = false;
function startAudio(){
  if (audioCtx) return;
  try {
    audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    const bufferSize = 2 * audioCtx.sampleRate;
    const buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
    const data = buffer.getChannelData(0);
    let lastOut = 0;
    for (let i=0;i<bufferSize;i++){
      const white = Math.random()*2-1;
      lastOut = (lastOut + 0.02*white) / 1.02;
      data[i] = lastOut * 4.2;
    }
    noiseSrc = audioCtx.createBufferSource();
    noiseSrc.buffer = buffer; noiseSrc.loop = true;
    const filter = audioCtx.createBiquadFilter();
    filter.type = 'lowpass'; filter.frequency.value = 700;
    gainNode = audioCtx.createGain(); gainNode.gain.value = 0.07;
    noiseSrc.connect(filter); filter.connect(gainNode); gainNode.connect(audioCtx.destination);
    noiseSrc.start();
  } catch(e){ /* audio not available, ignore */ }
}

function playPickupBlip(){

    const g = audioCtx.createGain();
    osc.type = 'sine';
    osc.frequency.setValueAtTime(560, audioCtx.currentTime);
    osc.frequency.exponentialRampToValueAtTime(920, audioCtx.currentTime + 0.14);
    g.gain.setValueAtTime(0.0001, audioCtx.currentTime);
    g.gain.exponentialRampToValueAtTime(0.16, audioCtx.currentTime + 0.02);
    g.gain.exponentialRampToValueAtTime(0.0001, audioCtx.currentTime + 0.32);
    osc.connect(g); g.connect(audioCtx.destination);
    osc.start(); osc.stop(audioCtx.currentTime + 0.34);
  } catch(e){ /* ignore */ }
}

    g.gain.exponentialRampToValueAtTime(0.0001, audioCtx.currentTime + 0.5);
    osc.connect(g); g.connect(audioCtx.destination);

  } catch(e){ /* ignore */ }
}

function smoothstep(x, a, b){
  const t = Math.min(1, Math.max(0, (x - a) / (b - a)));
  return t * t * (3 - 2 * t);
}
const moonDirTmp = new THREE.Vector3();
let dayFactor = 1, nightFactor = 0;
let lastTimeLabel = '';

function updateDayNight(delta){
  dayPhase = (dayPhase + delta / DAY_LENGTH) % 1;
  const angle = dayPhase * Math.PI * 2 - Math.PI / 2;
  const elevation = Math.sin(angle);
  const horiz = Math.cos(angle);
  sunDir.set(horiz * AZ_X, elevation, horiz * AZ_Z).normalize();
  sun.position.copy(sunDir).multiplyScalar(200);
  sun.target.position.set(0, 0, 0);
  sun.target.updateMatrixWorld();

  dayFactor = smoothstep(sunDir.y, -0.15, 0.25);
  nightFactor = 1 - dayFactor;
  const duskFactor = 1 - smoothstep(Math.abs(sunDir.y), 0.05, 0.4);

  // sky shader colors
  skyMat.uniforms.uTop.value.copy(SKY_NIGHT_TOP).lerp(SKY_DAY_TOP, dayFactor);
  skyMat.uniforms.uHorizon.value.copy(SKY_NIGHT_HORIZON).lerp(SKY_DAY_HORIZON, dayFactor).lerp(DUSK_TINT, duskFactor * 0.45);
  skyMat.uniforms.uHorizonLow.value.copy(SKY_NIGHT_HORIZONLOW).lerp(SKY_DAY_HORIZONLOW, dayFactor).lerp(DUSK_TINT, duskFactor * 0.55);
  skyMat.uniforms.uSunDir.value.copy(sunDir);
  skyMat.uniforms.uSunIntensity.value = Math.max(0.35, dayFactor * 1.6);

  // ocean colors
  oceanMat.uniforms.uDeep.value.copy(OCEAN_NIGHT_DEEP).lerp(OCEAN_DAY_DEEP, dayFactor);
  oceanMat.uniforms.uShallow.value.copy(OCEAN_NIGHT_SHALLOW).lerp(OCEAN_DAY_SHALLOW, dayFactor);
  oceanMat.uniforms.uFoam.value.copy(OCEAN_NIGHT_FOAM).lerp(OCEAN_DAY_FOAM, dayFactor);
  oceanMat.uniforms.uSunDir.value.copy(sunDir);

  scene.fog.color.copy(FOG_NIGHT).lerp(FOG_DAY, dayFactor);

  sun.intensity = 0.1 + dayFactor * 1.75;
  hemi.intensity = 0.22 + dayFactor * 0.85 + nightFactor * 0.12;
  fillLight.intensity = 0.12 + dayFactor * 0.3;
  moonLight.intensity = nightFactor * 0.5;
  renderer.toneMappingExposure = 0.78 + dayFactor * 0.38;

  sunSprite.position.copy(camera.position).addScaledVector(sunDir, SKY_DIST);
  sunSpriteMat.opacity = Math.max(0, dayFactor * 0.95 + duskFactor * 0.2);

  moonDirTmp.copy(sunDir).negate();
  moonGroup.position.copy(camera.position).addScaledVector(moonDirTmp, SKY_DIST);
  moonGroup.lookAt(camera.position);
  moonMat.opacity = 0.15 + nightFactor * 0.85;

  lampMat.emissiveIntensity = 1.1 + nightFactor * 1.3;

  let label, icon;
  if (dayPhase < 0.22) { label = 'Đêm khuya'; icon = '🌙'; }
  else if (dayPhase < 0.30) { label = 'Bình minh'; icon = '🌅'; }
  else if (dayPhase < 0.47) { label = 'Buổi sáng'; icon = '☀️'; }
  else if (dayPhase < 0.56) { label = 'Buổi trưa'; icon = '🌤️'; }
  else if (dayPhase < 0.70) { label = 'Buổi chiều'; icon = '🌇'; }
  else if (dayPhase < 0.78) { label = 'Hoàng hôn'; icon = '🌆'; }
  else { label = 'Ban đêm'; icon = '🌙'; }
  if (label !== lastTimeLabel){ timeText.textContent = label; timeIcon.textContent = icon; lastTimeLabel = label; }
}

  requestAnimationFrame(animate);
  const delta = Math.min(clock.getDelta(), 0.1);
  const t = clock.elapsedTime;

  // update shader times
  oceanMat.uniforms.uTime.value = t;
  oceanMat.uniforms.uCamPos.value.copy(camera.position);
  skyMat.uniforms.uTime.value = t;

  updateDayNight(delta);

  beaconGroup.rotation.y += delta * 0.55;
  beaconLight.intensity = (7 + nightFactor*7) + Math.sin(t*0.55)*2;

  fish.forEach(f=>{
    const u = f.userData;
    const tt = t*u.speed + u.phase;
    const arc = Math.sin(tt*2.6);
    f.position.x = u.center.x + Math.cos(tt)*u.radius;
    f.position.z = u.center.y + Math.sin(tt)*u.radius;
    f.position.y = u.yBase + Math.max(0, arc) * 1.35;
    f.rotation.y = -tt + Math.PI/2;
    f.rotation.z = Math.max(0,arc) * 0.5;
  });

  gulls.forEach(g=>{
    const u = g.userData;
    const tt = t*u.speed + u.phase;
    g.position.x = Math.cos(tt)*u.radius;
    g.position.z = Math.sin(tt)*u.radius + 4;
    g.position.y = u.height + Math.sin(tt*3)*0.6;
    g.rotation.y = -tt + Math.PI/2;
    g.userData.wing.rotation.z = Math.sin(t*8 + u.phase)*0.5;
  });

  cloudGroup.position.x = Math.sin(t*0.01)*10;

  updateMovement(delta);
  updateInteractables(t);
  updateDoor(delta);

  renderer.render(scene, camera);
}
animate();

window.addEventListener('resize', ()=>{
  camera.aspect = window.innerWidth/window.innerHeight;
  camera.updateProjectionMatrix();
  renderer.setSize(window.innerWidth, window.innerHeight);
});
